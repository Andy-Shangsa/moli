    # 单选按钮

    # 基础用法 若选项过多，建议使用 Select 选择器。

    <script>
export default {
  data() {
    return {
      model: "1"
    };
  }
};
</script>
<template>
  <ml-radio v-model="model" label="1">选项一</ml-radio>
  <ml-radio v-model="model" label="2">选项二</ml-radio>

  }

</script>
