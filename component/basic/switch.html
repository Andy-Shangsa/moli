    # 禁用状态

    <script>
export default {
  data() {
    return {
      value: false,
      value2: false,
      value3: false
    };
  }
};
</script>

<template>
  <ml-switch v-model="value" disabled></ml-switch>
  <ml-switch
    v-model="value2"
    active-color="#13ce66"
    inactive-color="#ff4949"
    disabled
  ></ml-switch>
  <ml-switch
    v-model="value3"
    active-color="#13ce66"
    inactive-color="#ff4949"
    active-value="100"
    inactive-value="0"
    disabled
  ></ml-switch>
</template>
